import java.util.Scanner;

public class Main {
    public static void main(String[] args) {

        double x;
        double y;
        double result;
        char operation;
        Scanner reader = new Scanner(System.in);
        System.out.print("¬ведите два числа: ");
        x = reader.nextDouble ();
        y = reader.nextDouble ();
        System.out.print("¬ведите операцию (+ , - , * , /): ");
        operation = reader.next().charAt(0);
        switch (operation) {
            case '+': result = x + y;
            break;
            case '-': result = x - y;
            break;
            case '*': result = x * y;
            break;
            case '/': result = x / y;
            break;
            default: System.out.printf("ќшибка. ¬ведите корректную операцию");
            return;
        }
        System.out.print("–езультат:  ");
        System.out.printf(x + "" + operation + "" + y + "=" + result);
    }
}
